19 // The following class defines a "synchronous" disk abstraction.
20 // As with other I/O devices, the raw physical disk is an asynchronous device --
21 // requests to read or write portions of the disk return immediately,
22 // and an interrupt occurs later to signal that the operation completed.
23 // (Also, the physical characteristics of the disk device assume that
24 // only one operation can be requested at a time).
25 //
26 // This class provides the abstraction that for any individual thread
27 // making a request, it waits around until the operation finishes before
28 // returning.
29 class SynchDisk {
30  public:
31  SynchDisk(char* name); // Initialize a synchronous disk,
32  // by initializing the raw Disk.
33  ~SynchDisk(); // De-allocate the synch disk data
34 
35  void ReadSector(int sectorNumber, char* data);
36  // Read/write a disk sector, returning
37  // only once the data is actually read
38  // or written. These call
39  // Disk::ReadRequest/WriteRequest and
40  // then wait until the request is done.
41  void WriteSector(int sectorNumber, char* data);
42 
43  void RequestDone(); // Called by the disk device interrupt
44  // handler, to signal that the
45  // current disk operation is complete.
46 
47  private:
48  Disk *disk; // Raw disk device
49  Semaphore *semaphore; // To synchronize requesting thread
50  // with the interrupt handler
51  Lock *lock; // Only one read/write request
52  // can be sent to the disk at a time
53 };
