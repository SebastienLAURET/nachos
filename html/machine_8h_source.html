1 // machine.h
2 // Data structures for simulating the execution of user programs
3 // running on top of Nachos.
4 //
5 // User programs are loaded into "mainMemory"; to Nachos,
6 // this looks just like an array of bytes. Of course, the Nachos
7 // kernel is in memory too -- but as in most machines these days,
8 // the kernel is loaded into a separate memory region from user
9 // programs, and accesses to kernel memory are not translated or paged.
10 //
11 // In Nachos, user programs are executed one instruction at a time,
12 // by the simulator. Each memory reference is translated, checked
13 // for errors, etc.

176 // NOTE: the hardware translation of virtual addresses in the user program
177 // to physical addresses (relative to the beginning of "mainMemory")
178 // can be controlled by one of:
179 // a traditional linear page table
180 // a software-loaded translation lookaside buffer (tlb) -- a cache of
181 // mappings of virtual page #'s to physical page #'s
182 //
183 // If "tlb" is NULL, the linear page table is used
184 // If "tlb" is non-NULL, the Nachos kernel is responsible for managing
185 // the contents of the TLB. But the kernel can use any data structure
186 // it wants (eg, segmented paging) for handling TLB cache misses.
187 //
188 // For simplicity, both the page table pointer and the TLB pointer are
189 // public. However, while there can be multiple page tables (one per address
190 // space, stored in memory), there is only one TLB (implemented in hardware).
191 // Thus the TLB pointer should be considered as *read-only*, although
192 // the contents of the TLB are free to be modified by the kernel software.

214 // Routines for converting Words and Short Words to and from the
215 // simulated machine's format of little endian. If the host machine
216 // is little endian (DEC and Intel), these end up being NOPs.
217 //
218 // What is stored in each format:
219 // host byte ordering:
220 // kernel data structures
221 // user registers
222 // simulated machine byte ordering:
223 // contents of main memory
224 //
225 // WARNING: the arguments to these macros are evaluated multiple times!
226 
227 #ifdef HOST_IS_BIG_ENDIAN
228 #define WordToHost(word) \
229  ((uint32_t)( \
230  (((uint32_t)(word) & (uint32_t)0x000000ffUL) << 24) | \
231  (((uint32_t)(word) & (uint32_t)0x0000ff00UL) << 8) | \
232  (((uint32_t)(word) & (uint32_t)0x00ff0000UL) >> 8) | \
233  (((uint32_t)(word) & (uint32_t)0xff000000UL) >> 24) ))
234 #define ShortToHost(sword) \
235  ((uint16_t)( \
236  (((uint16_t)(sword) & (uint16_t)0x00ffU) << 8) | \
237  (((uint16_t)(sword) & (uint16_t)0xff00U) >> 8) ));
238 #else
239 #define WordToHost(word) word
240 #define ShortToHost(sword) sword
